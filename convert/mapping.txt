METADATA (id, typ, parent)
(ILIAS2 Metadata --> ILIAS3 MetaData)

	meta
		id					-->		1.1 Identifier.Entry ("typ_id")
		inst				-->		1.1 Identifier.Catalog (ILIAS2 inst)
		typ					-->		1.1 Identifier.Entry | 1 General.Structure .AggregationLevel *** (Not precise)
		title				-->		1.2 Title
		lang				-->		1.3 Language
		description			-->		1.4 Description
		diff				-->		5 Educational.Difficulty
		level				-->		5 Educational.Context
		status				-->		2 Lifecycle.Status
		material_level		-->		9 Classification (.Purpose = "EducationalLevel") -> 9.2.2 Taxon
		last_modified_date	-->		2.3 Contribute (.Role = "Author" | "TechnicalImplementer") -> 2.3.3 Date
		publisher			-->		2.3 Contribute (.Role = "Publisher") -> 2.3.2 Entity *** (only typ="le")
		publish_date		-->		2.3 Contribute (.Role = "Publisher") -> 2.3.3 Date *** (only typ="le")

Changes to the DTD:
-------------------

Paragraph -> new attribute:	Characteristic (Example | Citation | Mnemonic) #IMPLIED
Contribute -> attribute Role: double entry for validator
Title -> (%titletext;)* --> (#PCDATA)
Description -> (%simpletext;)* --> (#PCDATA)
Taxon -> (Entry) --> (#PCDATA), new attribute: Language (%langcode;) #REQUIRED
Entry -> removed (not needed at all)
Educational -> Language+ removed (obsolete)
Definition -> (Paragraph+) --> (Paragraph | LearningObject)+
Answer -> new attribute: Solution (Right | Wrong) #IMPLIED
Paragraph -> extended attribute: Characteristic (Headline | Example | Citation | Mnemonic) #IMPLIED
Content -> (Paragraph | LearningObject) --> (Paragraph | LearningObject)
Text -> removed (not needed at all)
Headline -> removed (not needed at all)
TableData -> new attribute: Width CDATA #IMPLIED
operatingsystem -> removed #PCDATA from entity
browser -> removed #PCDATA from entity
LearningObject -> new Element:	<!ELEMENT Parameter (ParameterName, ParameterValue)+>
								<!ELEMENT ParameterName (#PCDATA)>
								<!ELEMENT ParameterValue (#PCDATA)>
operatingsystem -> replaced entitiy
browser -> replaced entitiy
Layout -> new value for attribute: TargetFrame (... | NEW)
